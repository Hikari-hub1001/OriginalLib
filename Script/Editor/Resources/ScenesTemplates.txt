#if UNITY_EDITOR
using UnityEditor;
using UnityEditor.SceneManagement;
#endif

/// <summary>
/// Scenes（自動生成クラス）。 
/// </summary> 
public enum Scenes
{#ScenesEnum#}

#if UNITY_EDITOR
/// <summary>
/// メニューバーからシーン遷移
/// </summary> 
class MenuScene
{
	static string title = "Scene(s) Have Been Modified";
	static string message1 = "Do you want to save the changes you made in the scenes:\n";
	static string message2 = "\n\nYour changes will be lost if you don't save them.";

	#MenuMethodsStart#
	[MenuItem("Scenes/#ScenePath#")]
	static void Menu#SceneName#()
	{
		if(DirtyCheck())
		{
			EditorSceneManager.OpenScene("#SceneFullPath#");
		}
	}

	#MenuMethodsEnd#

	static bool DirtyCheck()
	{
		if (EditorSceneManager.GetActiveScene().isDirty)
		{
			int option = EditorUtility.DisplayDialogComplex(title, message1 + EditorSceneManager.GetActiveScene().path + message2
				, "Save", "Don't Save", "Cancel");
			switch (option)
			{
				case 0:
					EditorSceneManager.SaveScene(EditorSceneManager.GetActiveScene());
					break;
				case 2:
					return false;
			}
		}

		return true;
	}

}
#endif
